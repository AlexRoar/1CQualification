32 STree::go(STree::AlgoPos st, long long int l, long long int r) {
33 while (l < r)
34 if (st.pos == nodes[st.v].len()) {
35 st = AlgoPos(nodes[st.v].get(text(l)), 0);
36 if (st.v == -1) return st;

38 if (text(nodes[st.v].start + st.pos) != text(l))

48 long long STree::split(STree::AlgoPos st) {
49 if (st.pos == nodes[st.v].len())
50 return st.v;
51 if (st.pos == 0)
52 return nodes[st.v].parent;
53 STreeNode v = nodes[st.v];
54 long long id = nodes.size();
55 nodes.emplace_back(v.start, v.start + st.pos, v.parent);
56 nodes[v.parent].get(text(v.start)) = id;
57 nodes[id].get(text(v.start + st.pos)) = st.v;
58 nodes[st.v].parent = id;
59 nodes[st.v].start += st.pos;
60 return id;
61 }

136 st.pos = nodes[st.v].len();
